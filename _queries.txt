mutation DeleteMutation($id: ID!){
  deleteGame(id: $id)
  {
    id,
    title,
    platform
  }
}

-----

mutation AddMutation($game: AddGameInput){

  {
    id,
    title,
    platform
  }
}

---

  {
    title,
    platform
  }
}

variables:
{
  "edits": {

  },
